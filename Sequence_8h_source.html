394 endSequenceShouldReturnTrueBetweenLoops = false;

396 sequenceStep += 1;
397 timesToLoop = times;
398 // timesLooped should be incremented before the check
399 timesLooped += 1;
400 if (timesLooped == timesToLoop)
401 {
402 endSequenceShouldReturnTrueBetweenLoops = true;
403 // This will allow the sequence to restart from the first step
404 // and simply return true after the specified number of times has been looped
405 // This way there is a queue for when the fixed number of loops has been completed
406 // and the sequence can be restarted from the first step just like with a normal
407 // loop. This results in reusable sequences that do not require a manual restart.
408 shouldLoop = restartAfterCompletion;
409 timesToLoop = 0;
410 timesLooped = 0;
411 }
412 }
413
414 checkedSteps += 1;
415 return *this;
